# TO RUN THIS:
- create an environment and activate it
-  pip install openai
-  pip install sentence-transformers
-  pip install -r requirements.txt
-  $env:OPENAI_API_KEY= "... api key here....."
-  python app.py
-  go to index.html and run in five server
